Frontend Development Evolution

Building the same Todo App across different eras of web development

üåê The Beginning: HTML + CSS + Vanilla JavaScript

The early web was simple but powerful. Direct DOM manipulation, inline event handlers, and imperative programming were the norm. Every interaction required manual DOM updates.

üìÑ HTML + CSS + Vanilla JavaScript (1995-2005)
// Early web development approach
<div id="todoApp">
  <h3>Todo List</h3>
  <input type="text" id="todoInput" 
         placeholder="Enter task...">
  <button onclick="addTodo()">Add</button>
  <ul id="todoList"></ul>
</div>

<script>
var todos = [];
var todoCounter = 0;

function addTodo() {
  var input = document.getElementById('todoInput');
  var text = input.value.trim();
  
  if (text === '') return;
  
  var todo = {
    id: ++todoCounter,
    text: text,
    completed: false
  };
  
  todos.push(todo);
  input.value = '';
  renderTodos();
}

function toggleTodo(id) {
  for (var i = 0; i < todos.length; i++) {
    if (todos[i].id === id) {
      todos[i].completed = !todos[i].completed;
      break;
    }
  }
  renderTodos();
}

function deleteTodo(id) {
  for (var i = 0; i < todos.length; i++) {
    if (todos[i].id === id) {
      todos.splice(i, 1);
      break;
    }
  }
  renderTodos();
}

function renderTodos() {
  var list = document.getElementById('todoList');
  list.innerHTML = ''; // Clear existing items
  
  for (var i = 0; i < todos.length; i++) {
    var todo = todos[i];
    var li = document.createElement('li');
    li.className = todo.completed ? 'completed' : '';
    
    li.innerHTML = 
      '<input type="checkbox" ' +
      'onchange="toggleTodo(' + todo.id + ')" ' +
      (todo.completed ? 'checked' : '') + '>' +
      '<span>' + todo.text + '</span>' +
      '<button onclick="deleteTodo(' + todo.id + ')">Delete</button>';
    
    list.appendChild(li);
  }
}
</script>

Todo List (Vanilla JS)

Pros

  • Simple and straightforward
  • No external dependencies
  • Fast loading
  • Direct control over DOM

Cons

  • Lots of boilerplate code
  • Manual DOM manipulation
  • Browser compatibility issues
  • Mixing HTML and JavaScript
  • Hard to maintain as it grows

üíö The jQuery Revolution

jQuery simplified DOM manipulation and AJAX calls. "Write less, do more" became the motto. Cross-browser compatibility was finally manageable, and animations became easy.

<script>
$(document).ready(function() {
  var todos = [];
  var todoCounter = 0;

  // Event delegation for dynamic elements
  $('#addBtn').on('click', addTodo);
  $('#jqueryTodoInput').on('keypress', function(e) {
    if (e.which === 13) addTodo();
  });
  
  // Event delegation for dynamic content
  $('#jqueryTodoList').on('change', '.todo-checkbox', function() {
    var id = parseInt($(this).data('id'));
    toggleTodo(id);
  });
  
  $('#jqueryTodoList').on('click', '.delete-btn', function() {
    var id = parseInt($(this).data('id'));
    deleteTodo(id);
  });

  function addTodo() {
    var text = $('#jqueryTodoInput').val().trim();
    
    if (text === '') return;
    
    var todo = {
      id: ++todoCounter,
      text: text,
      completed: false
    };
    
    todos.push(todo);
    $('#jqueryTodoInput').val('');
    renderTodos();
  }

  function toggleTodo(id) {
    var todo = $.grep(todos, function(t) { 
      return t.id === id; 
    })[0];
    
    if (todo) {
      todo.completed = !todo.completed;
      renderTodos();
    }
  }

  function deleteTodo(id) {
    todos = $.grep(todos, function(t) { 
      return t.id !== id; 
    });
    renderTodos();
  }

  function renderTodos() {
    var $list = $('#jqueryTodoList');
    $list.empty(); // jQuery's elegant DOM clearing
    
    $.each(todos, function(index, todo) {
      var $li = $('<li>')
        .addClass('todo-item')
        .toggleClass('completed', todo.completed);
      
      var $checkbox = $('<input>')
        .attr('type', 'checkbox')
        .addClass('todo-checkbox')
        .data('id', todo.id)
        .prop('checked', todo.completed);
      
      var $text = $('<span>').text(todo.text);
      
      var $deleteBtn = $('<button>')
        .addClass('delete-btn')
        .data('id', todo.id)
        .text('Delete');
      
      $li.append($checkbox, $text, $deleteBtn);
      $list.append($li);
      
      // jQuery animations
      $li.hide().fadeIn(300);
    });
  }
});
</script>

// Model - represents a single todo item
var Todo = Backbone.Model.extend({
  defaults: {
    text: '',
    completed: false
  },
  
  toggle: function() {
    this.set('completed', !this.get('completed'));
  }
});

// Collection - manages multiple todos
var TodoCollection = Backbone.Collection.extend({
  model: Todo,
  
  completed: function() {
    return this.filter(function(todo) {
      return todo.get('completed');
    });
  },
  
  remaining: function() {
    return this.without.apply(this, this.completed());
  }
});

// View for individual todo items
var TodoView = Backbone.View.extend({
  tagName: 'li',
  className: 'todo-item',
  
  events: {
    'change .todo-checkbox': 'toggleCompleted',
    'click .delete-btn': 'deleteTodo'
  },
  
  initialize: function() {
    this.listenTo(this.model, 'change', this.render);
    this.listenTo(this.model, 'destroy', this.remove);
  },
  
  render: function() {
    var completed = this.model.get('completed');
    
    this.$el.html(
      '<input type="checkbox" class="todo-checkbox" ' + 
      (completed ? 'checked' : '') + '>' +
      '<span>' + this.model.get('text') + '</span>' +
      '<button class="delete-btn">Delete</button>'
    );
    
    this.$el.toggleClass('completed', completed);
    return this;
  },
  
  toggleCompleted: function() {
    this.model.toggle();
  },
  
  deleteTodo: function() {
    this.model.destroy();
  }
});

// Main application view
var AppView = Backbone.View.extend({
  el: '#backboneTodoApp',
  
  events: {
    'click #backboneAddBtn': 'addTodo',
    'keypress #backboneTodoInput': 'addOnEnter'
  },
  
  initialize: function() {
    this.todos = new TodoCollection();
    this.listenTo(this.todos, 'add', this.addOne);
    this.listenTo(this.todos, 'reset', this.addAll);
  },
  
  addTodo: function() {
    var text = this.$('#backboneTodoInput').val().trim();
    
    if (text === '') return;
    
    this.todos.create({
      text: text,
      completed: false
    });
    
    this.$('#backboneTodoInput').val('');
  },
  
  addOnEnter: function(e) {
    if (e.which === 13) this.addTodo();
  },
  
  addOne: function(todo) {
    var view = new TodoView({ model: todo });
    this.$('#backboneTodoList').append(view.render().el);
  },
  
  addAll: function() {
    this.todos.each(this.addOne, this);
  }
});

// Initialize the app
var app = new AppView();
</script>

<script type="text/babel">
// Individual Todo Item Component
const TodoItem = ({ todo, onToggle, onDelete }) => {
  return (
    <li className={`todo-item ${todo.completed ? 'completed' : ''}`}>
      <input
        type="checkbox"
        checked={todo.completed}
        onChange={() => onToggle(todo.id)}
      />
      <span>{todo.text}</span>
      <button
        className="delete-btn"
        onClick={() => onDelete(todo.id)}
      >
        Delete
      </button>
    </li>
  );
};

// Main Todo App Component
const TodoApp = () => {
  // State management with hooks
  const [todos, setTodos] = React.useState([]);
  const [inputValue, setInputValue] = React.useState('');
  const [nextId, setNextId] = React.useState(1);

  // Add new todo
  const addTodo = () => {
    if (inputValue.trim() === '') return;
    
    const newTodo = {
      id: nextId,
      text: inputValue.trim(),
      completed: false
    };
    
    setTodos(prevTodos => [...prevTodos, newTodo]);
    setNextId(prev => prev + 1);
    setInputValue('');
  };

  // Toggle todo completion
  const toggleTodo = (id) => {
    setTodos(prevTodos =>
      prevTodos.map(todo =>
        todo.id === id 
          ? { ...todo, completed: !todo.completed }
          : todo
      )
    );
  };

  // Delete todo
  const deleteTodo = (id) => {
    setTodos(prevTodos => 
      prevTodos.filter(todo => todo.id !== id)
    );
  };

  // Handle Enter key
  const handleKeyPress = (e) => {
    if (e.key === 'Enter') {
      addTodo();
    }
  };

  // Render component
  return (
    <div className="todo-app">
      <div className="todo-header">
        <h3>Todo List (React)</h3>
      </div>
      
      <div className="todo-list">
        <input
          type="text"
          className="todo-input"
          placeholder="Enter task..."
          value={inputValue}
          onChange={(e) => setInputValue(e.target.value)}
          onKeyPress={handleKeyPress}
        />
        
        <button className="add-btn" onClick={addTodo}>
          Add Task
        </button>
        
        <ul>
          {todos.map(todo => (
            <TodoItem
              key={todo.id}
              todo={todo}
              onToggle={toggleTodo}
              onDelete={deleteTodo}
            />
          ))}
        </ul>
        
        {todos.length === 0 && (
          <p style={{textAlign: 'center', color: '#888', marginTop: '20px'}}>
            No tasks yet. Add one above!
          </p>
        )}
        
        {todos.length > 0 && (
          <p style={{textAlign: 'center', color: '#666', marginTop: '10px'}}>
            {todos.filter(t => !t.completed).length} of {todos.length} tasks remaining
          </p>
        )}
      </div>
    </div>
  );
};

// Render the app
ReactDOM.render(<TodoApp />, document.getElementById('reactTodoApp'));
</script>
